1. setup `.env`

    HOST=localhost
    PORT=5000

    PGHOST=localhost
    PGUSER=

    PGPASSWORD=
    PGPORT=5432

    # TOKENIZE
    ACCESS_TOKEN_KEY=
    REFRESH_TOKEN_KEY=
    ACCCESS_TOKEN_AGE=
